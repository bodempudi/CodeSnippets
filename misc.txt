    IF (
        (CALCULATE([SUM_Sales],Dimdate[WeekOfYear]=week) = BLANK ()),
        MAXX (
            FILTER ( ALL ( Dimdate ), Dimdate[WeekOfYear] < MAX ( Dimdate[WeekOfYear] ) ),
            [WeekIndex]
        ),
        MAX ( Dimdate[WeekOfYear] )
    )
RETURN

        [SUM_Sales],
        FILTER ( ALL ( Dimdate ), Dimdate[WeekOfYear] = LastWeek_notnullsales )
    )
